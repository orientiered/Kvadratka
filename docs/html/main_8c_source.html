193quadraticEquation_t* solveQuadratic(quadraticEquation_t* equation) {
194 double a = equation->a,
195 b = equation->b,
196 c = equation->c;
197
198 assert(isfinite(a)); //checking input for NaNs
199 assert(isfinite(b));
200 assert(isfinite(c));
201
202 if (fabs(a) < EPSILON) { //checking for zeros in coefficients; we divide only by a, so this check is essential
203 //a = 0
204 if (fabs(b) < EPSILON) {
205 //b = 0
206 if (fabs(c) < EPSILON) { // 0 = 0
207 equation->answer.code = INF_ROOTS;
208 } else {
209 printf("Equation can't be solved\n");
210 equation->answer.code = ZERO_ROOTS;
211 }
212 } else {
213 equation->answer.code = ONE_ROOT;
214 equation->answer.x1 = -c/b;
215 }
216 } else {
217 double D = b*b - 4*a*c; //calculating discriminant
218 if (fabs(D) < EPSILON) { //D = 0
219 equation->answer.code = ONE_ROOT;
220 equation->answer.x1 = -b / (2*a);
221 } else if (D < 0) {
222 equation->answer.code = ZERO_ROOTS; //D < 0
223 printf("Equation can't be solved in R: D = %g < 0\n", D);
224 } else { //D > 0
225 equation->answer.code = TWO_ROOTS;
226 double D_sqrt = sqrt(D);
227 equation->answer.x1 = (-b + D_sqrt)/(2*a);
228 equation->answer.x2 = (-b - D_sqrt)/(2*a);
229 }
230 }
231 //fix -0 case
232 equation->answer.x1 = fixMinusZero(equation->answer.x1);
233 equation->answer.x2 = fixMinusZero(equation->answer.x2);
234 return equation;
235}
